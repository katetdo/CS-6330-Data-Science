flights = pd.read_feather('flights.feather')

# adding elapsed days to flight
first_date = flights['date'][0]
flights['elapsed_days'] = flights['date'].apply(lambda d: (d-first_date).days)

# making a deep copy of flights to not give it new/random columns
new_data = flights.copy(True)
new_data['set'] = np.where(new_data['date'].dt.year < 2017, "Train", "Test")
new_data['ave'] = new_data['count'].rolling(7).mean()

# using count and elapsed days for linregress
count = new_data['count']
elapsed_days = new_data['elapsed_days']
results = stats.linregress(elapsed_days, count)

# creating prediction line
pred = elapsed_days * results.slope + results.intercept
new_data['prediction'] = pred
new_data

sns.lineplot(data = new_data, x = 'elapsed_days', y = 'ave', hue = 'set')
plt.plot(elapsed_days, new_data['prediction'], 'r' ,label = 'fitted line')
plt.title('Total Number of Flights Each Day From 2013-2017')

flights['season'] = flights['season'].astype('category')
flights['day_of_week'] = flights['day_of_week'].astype('category')

y, X = dmatrices("count ~ prcp + snow + tmax + tmin + holiday + elapsed_days + season + day_of_week - 1", data = flights, return_type = 'dataframe')

model = sm.OLS(y,X)
results = model.fit()
pred_y = results.predict()
results.summary()

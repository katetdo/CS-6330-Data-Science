#sort houses into bins then separate into train and test
train = pd.DataFrame()
test = pd.DataFrame()

for x in range(0, houses["bin"].max()): # loop through the number of bins
    sort_bin = houses[houses["bin"] == x] # df with houses in said bin
    
    for index, row in sort_bin.iterrows():
        rand = random.random()
        
        if rand <= 0.25: #(add to test set)
            test = pd.concat([test, row.to_frame().T])
        else: #(add to train set) 
            train = pd.concat([train, row.to_frame().T])
